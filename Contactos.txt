Contactos:
Pantalla de visualización de todos los Contactos del Teléfono utilizando la FlatList.
Se debe mostrar el Nombre y Apellido / Número telefónico y un indic


import React, { useState, useEffect } from 'react';
import { View, Text, FlatList } from 'react-native';
import Contacts from 'react-native-contacts';

const Contactos = () => {
  const [contactos, setContactos] = useState([]);

  useEffect(() => {
    Contacts.getAll().then((contactos) => {
      setContactos(contactos);
    });
  }, []);
  const renderItem = ({ item }) => (
    <View>
      <Text>{item.givenName} {item.familyName}</Text>
      <Text>{item.phoneNumbers[0].number}</Text>
      {item.isDefaultForEmergency && <Icon name="emergency" />}
    </View>
  );

  return (
    <FlatList
      data={contactos}
      renderItem={renderItem}
      keyExtractor={(item) => item.recordID}
    />
  );
};


export default Contactos;
